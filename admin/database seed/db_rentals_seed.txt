SET NAMES utf8mb4;

CREATE DATABASE IF NOT EXISTS ymzm
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE ymzm;

-- =========================
-- Tables
-- =========================

CREATE TABLE IF NOT EXISTS rental_groups (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  group_key VARCHAR(80) NOT NULL UNIQUE,
  group_name VARCHAR(120) NOT NULL,
  description TEXT NULL,
  cover_img VARCHAR(255) NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS rental_items (
  id VARCHAR(40) NOT NULL PRIMARY KEY,
  group_id INT UNSIGNED NOT NULL,
  item_category VARCHAR(120) NOT NULL,
  name VARCHAR(160) NOT NULL,
  price DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  img VARCHAR(255) NULL,
  stock INT NOT NULL DEFAULT 0,
  default_qty INT NOT NULL DEFAULT 1,
  is_active TINYINT(1) NOT NULL DEFAULT 1,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_rental_items_group
    FOREIGN KEY (group_id) REFERENCES rental_groups(id)
    ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS rental_item_colors (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  item_id VARCHAR(40) NOT NULL,
  color_name VARCHAR(60) NOT NULL,
  color_hex VARCHAR(20) NOT NULL,
  color_stock INT NOT NULL DEFAULT 0,
  CONSTRAINT fk_rental_item_colors_item
    FOREIGN KEY (item_id) REFERENCES rental_items(id)
    ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
