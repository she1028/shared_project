SET NAMES utf8mb4;

CREATE DATABASE IF NOT EXISTS ymzm
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE ymzm;

CREATE TABLE IF NOT EXISTS packages (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  category ENUM('wedding','children','debut','corporate') NOT NULL,
  slug VARCHAR(80) NOT NULL,
  package_title VARCHAR(100) NOT NULL,
  package_name VARCHAR(150) NOT NULL,
  description TEXT NOT NULL,
  note TEXT NULL,
  starts_at DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  image_path VARCHAR(255) NULL,
  is_active TINYINT(1) NOT NULL DEFAULT 1,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY uq_packages_category_slug (category, slug)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
