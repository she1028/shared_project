SET NAMES utf8mb4;

CREATE DATABASE IF NOT EXISTS ymzm
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE ymzm;

-- Stores the per-package inclusion “header” data you have in inclusion.js
-- (offer/title/price/note/image). This does NOT change your existing packages table.
CREATE TABLE IF NOT EXISTS package_details (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  category ENUM('wedding','children','debut','corporate') NOT NULL,
  slug VARCHAR(80) NOT NULL,
  offer VARCHAR(50) NOT NULL,
  title VARCHAR(150) NOT NULL,
  price_label VARCHAR(50) NOT NULL,
  note TEXT NULL,
  image_path VARCHAR(255) NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY uq_package_details_category_slug (category, slug)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Stores the lists (menu/rentals/decorations/services)
CREATE TABLE IF NOT EXISTS package_inclusion_items (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  package_detail_id INT UNSIGNED NOT NULL,
  inclusion_type ENUM('menu','rentals','decorations','services') NOT NULL,
  item_text VARCHAR(255) NOT NULL,
  sort_order INT UNSIGNED NOT NULL DEFAULT 0,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  KEY idx_inclusion_lookup (package_detail_id, inclusion_type, sort_order),
  CONSTRAINT fk_package_inclusions_detail
    FOREIGN KEY (package_detail_id) REFERENCES package_details(id)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
